//=========================================
//  1. Class and Object (Student Program)
//=========================================

#include <iostream>
using namespace std;

class Student {
public:
    string name;
    int roll;

    void input() {
        cout << "Enter Name: ";
        cin >> name;
        cout << "Enter Roll: ";
        cin >> roll;
    }

    void display() {
        cout << "Name: " << name << endl;
        cout << "Roll: " << roll;
    }
};

int main() {
    Student s;
    s.input();
    s.display();
    return 0;
}

//=========================================
//  3. Copy Constructor
//=========================================

#include <iostream>
using namespace std;

class Test {
    int a;

public:
    Test(int x) {
        a = x;
    }

    Test(Test &t) {
        a = t.a;
    }

    void show() {
        cout << a;
    }
};

int main() {
    Test t1(10);
    Test t2 = t1;
    t2.show();
    return 0;
}

//=========================================
//  5. Function Overloading
//=========================================

#include <iostream>
using namespace std;

class Sum {
public:
    void add(int a, int b) {
        cout << a + b << endl;
    }

    void add(float a, float b) {
        cout << a + b;
    }
};

int main() {
    Sum s;
    s.add(4, 5);
    s.add(2.3f, 3.2f);
    return 0;
}


//=========================================
//  6. Operator Overloading
//=========================================

#include <iostream>
using namespace std;

class Demo {
    int x;

public:
    Demo(int a) {
        x = a;
    }

    Demo operator+(Demo d) {
        return Demo(x + d.x);
    }

    void show() {
        cout << x;
    }
};

int main() {
    Demo d1(5), d2(7);
    Demo d3 = d1 + d2;
    d3.show();
    return 0;
}

//=========================================
//  9. Multiple Inheritance
//=========================================

#include <iostream>
using namespace std;

class A {
public:
    void showA() {
        cout << "A\n";
    }
};

class B {
public:
    void showB() {
        cout << "B";
    }
};

class C : public A, public B {
};

int main() {
    C obj;
    obj.showA();
    obj.showB();
    return 0;
}


//=========================================
//  10. Virtual Function
//=========================================

#include <iostream>
using namespace std;

class Base {
public:
    virtual void show() {
        cout << "Base";
    }
};

class Derived : public Base {
public:
    void show() {
        cout << "Derived";
    }
};

int main() {
    Base *ptr;
    Derived d;
    ptr = &d;
    ptr->show();
    return 0;
}


//=========================================

    return 0;
}


//=========================================
//  12. Exception Handling
//=========================================

#include <iostream>
using namespace std;

int main() {
    try {
        int a = 10, b = 0;
        if (b == 0)
            throw b;
        cout << a / b;
    } catch (int x) {
        cout << "Divide by zero error";
    }
    return 0;
}

//=========================================
//  15. Friend Function
//=========================================

using namespace std;

    friend void show(Demo);
};

}

    Demo d;
    show(d);
    return 0;
}
